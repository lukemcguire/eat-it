    """Ingredient combiner service for shopping list generation."""
    from dataclasses import dataclass
    from collections import defaultdict
    from ingredient_parser import parse_ingredient


    @dataclass
    class CombinedIngredient:
        """A combined ingredient with normalized name and summed quantity."""
        name: str
        quantity: float | None
        unit: str | None
        raw_strings: list[str]


    def _normalize_name(name: str) -> str:
        """Normalize ingredient name for combining.

        - Lowercase
        - Strip whitespace
        - Remove preparation notes (text after comma or in parens)
        """
        # Remove preparation notes
        name = name.split(",")[0]
        name = name.split("(")[0]
        return name.lower().strip()


    def combine_ingredients(ingredient_strings: list[str]) -> list[CombinedIngredient]:
        """Combine ingredients from multiple recipes.

        - Normalizes names (lowercase, stripped, prep removed)
        - Sums quantities for same ingredient+unit
        - Keeps different ingredients separate
        - Keeps same ingredient with different units separate

        Args:
            ingredient_strings: List of raw ingredient strings

        Returns:
            List of CombinedIngredient sorted by name
        """
        combined = defaultdict(
            lambda: {"quantity": 0.0, "unit": None, "raw": [], "name": None}
        )

        for raw in ingredient_strings:
            result = parse_ingredient(raw, string_units=True)

            # Get normalized name (ignore preparation)
            if result.name:
                name = _normalize_name(result.name[0].text)
            else:
                name = raw.lower().strip()

            if result.amount:
                qty = float(result.amount[0].quantity) if result.amount[0].quantity else 0
                unit = result.amount[0].unit if isinstance(
                    result.amount[0].unit, str
                ) else str(result.amount[0].unit) if result.amount[0].unit else None
                key = f"{name}|{unit}"
                combined[key]["quantity"] += qty
                combined[key]["unit"] = unit
            else:
                key = f"{name}|"
                combined[key]["quantity"] = 0.0

            combined[key]["name"] = name
            combined[key]["raw"].append(raw)

        # Convert to CombinedIngredient list
        result_list = [
            CombinedIngredient(
                name=v["name"],
                quantity=v["quantity"] if v["quantity"] > 0 else None,
                unit=v["unit"],
                raw_strings=v["raw"]
            )
            for v in combined.values()
        ]

        # Sort by name
        return sorted(result_list, key=lambda x: x.name)
    ```

    """Section categorizer service for auto-categorizing ingredients."""
    from sqlmodel import Session, select
    from eat_it.models.store_section import StoreSection


    DEFAULT_SECTIONS = [
        ("Produce", [
            "onion", "garlic", "tomato", "lettuce", "carrot", "pepper",
            "celery", "potato", "spinach", "broccoli", "cucumber", "zucchini",
            "mushroom", "cilantro", "parsley", "lemon", "lime", "apple",
            "banana", "orange", "avocado", "cabbage", "kale", "beans", "corn"
        ]),
        ("Dairy", [
            "milk", "cheese", "butter", "cream", "yogurt", "egg",
            "sour cream", "cottage", "ricotta", "mozzarella", "cheddar",
            "parmesan"
        ]),
        ("Meat", [
            "chicken", "beef", "pork", "bacon", "sausage", "fish",
            "salmon", "turkey", "ham", "lamb", "shrimp", "ground"
        ]),
        ("Bakery", [
            "bread", "tortilla", "bagel", "roll", "baguette", "pita",
            "croissant", "muffin"
        ]),
        ("Pantry", [
            "flour", "sugar", "salt", "oil", "vinegar", "rice", "pasta",
            "sauce", "broth", "stock", "soy sauce", "ketchup", "mustard",
            "mayo", "honey", "vanilla", "baking", "yeast", "cornstarch",
            "breadcrumb"
        ]),
        ("Frozen", [
            "frozen", "ice cream"
        ]),
    ]


    def categorize_ingredient(name: str) -> str:
        """Categorize an ingredient to a store section.

        Uses case-insensitive substring matching against DEFAULT_SECTIONS.

        Args:
            name: The ingredient name

        Returns:
            Section name (e.g., "Produce", "Dairy") or "Other" if no match
        """
        name_lower = name.lower()
        for section_name, keywords in DEFAULT_SECTIONS:
            if any(kw in name_lower for kw in keywords):
                return section_name
        return "Other"


    def get_or_create_section_id(session: Session, section_name: str) -> int:
        """Get or create a section ID by name.

        Args:
            session: Database session
            section_name: Name of the section

        Returns:
            The section ID
        """
        section = session.exec(
            select(StoreSection).where(StoreSection.name == section_name)
        ).first()

        if section:
            return section.id

        # Create new section with next sort_order
        max_order = session.exec(
            select(StoreSection.sort_order)
        ).all()
        next_order = max(o[0] for o in max_order) + 1 if max_order else 100

        new_section = StoreSection(
            name=section_name,
            sort_order=next_order,
            is_default=False
        )
        session.add(new_section)
        session.commit()
        session.refresh(new_section)
        return new_section.id
    ```
