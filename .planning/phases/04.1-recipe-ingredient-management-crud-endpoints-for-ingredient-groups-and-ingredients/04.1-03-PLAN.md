class TestCreateGroup:

class TestListGroups:
- test_list_groups_empty: Create recipe, GET groups, assert []
- test_list_groups: Create recipe + 2 groups, GET, assert both returned
- test_list_groups_recipe_not_found: GET /recipes/999/groups, assert 404

class TestUpdateGroup:
- test_update_group: Create recipe + group, PATCH with new name, assert updated
- test_update_group_not_found: PATCH /recipes/1/groups/999, assert 404
- test_update_group_recipe_not_found: PATCH /recipes/999/groups/1, assert 404

class TestDeleteGroup:
- test_delete_group: Create recipe + group, DELETE, assert 204
- test_delete_group_removes_ingredients: Create recipe + group + ingredient,
  DELETE group, verify ingredient gone (try to GET /ingredients)
- test_delete_group_not_found: DELETE /recipes/1/groups/999, assert 404

class TestCreateIngredient:
- test_create_ingredient: Create recipe + group, POST ingredient with group_id,
  assert 201
- test_create_ingredient_auto_default_group: Create recipe (no group), POST
  ingredient without group_id, assert group auto-created with name "Ingredients"
- test_create_ingredient_recipe_not_found: POST /recipes/999/ingredients, assert 404

class TestListIngredients:
- test_list_ingredients_empty: Create recipe, GET ingredients, assert []
- test_list_ingredients: Create recipe + 2 groups + 3 ingredients, GET, assert
  all 3 returned ordered by group_id, display_order
- test_list_ingredients_recipe_not_found: GET /recipes/999/ingredients, assert 404

class TestUpdateIngredient:
- test_update_ingredient: Create recipe + group + ingredient, PATCH with new
  name/quantity, assert updated
- test_update_ingredient_not_found: PATCH /recipes/1/ingredients/999, assert 404
- test_update_ingredient_recipe_not_found: PATCH /recipes/999/ingredients/1,
  assert 404

class TestDeleteIngredient:
- test_delete_ingredient: Create recipe + group + ingredient, DELETE, assert 204
- test_delete_ingredient_removes_empty_group: Create recipe + group + 1 ingredient,

class TestBulkReplace:
