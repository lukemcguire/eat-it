    The endpoint should:

    1. Route: GET /search
       - Query parameter: q (required, min_length=1)

       - Returns: list[RecipePublic]

    2. Search logic (per CONTEXT.md and RESEARCH.md):
       ```python
       @router.get("/search", response_model=list[RecipePublic])
       def search_recipes(
           *,
           session: Session = Depends(get_session),
           request: Request,
           q: str = Query(..., min_length=1, description="Search query"),
           limit: int = Query(20, ge=1, le=100),
       ) -> list[Recipe]:
           results: list[Recipe] = []

           # 1. Try semantic search if model available
           if request.app.state.embedding_model:
               try:
                   query_embedding = generate_embedding(q, request.app.state.embedding_model)
                   query_binary = serialize_f32(query_embedding)

                   conn = session.connection().connection.dbapi_connection
                   try:
                       rows = conn.execute(
                           """
                           SELECT rowid as recipe_id, distance
                           FROM recipe_embeddings
                           WHERE embedding MATCH ?
                           ORDER BY distance
                           LIMIT ?
                           """,
                           [query_binary, limit]
                       ).fetchall()

                       if rows:
                           # Filter by distance threshold (L2 distance)
                           THRESHOLD = 1.5  # Tune based on testing
                           recipe_ids = [r[0] for r in rows if r[1] < THRESHOLD]

                           if recipe_ids:
                               # Fetch recipes preserving KNN order
                               id_to_distance = {r[0]: r[1] for r in rows if r[1] < THRESHOLD}
                               recipes = session.exec(
                                   select(Recipe).where(Recipe.id.in_(recipe_ids))
                               ).all()
                               results = sorted(recipes, key=lambda r: id_to_distance.get(r.id, float('inf')))
                   finally:
                       conn.close()
               except Exception:
                   pass  # Fall through to keyword search

           # 2. Fallback to keyword search
           if not results:
               results = list(session.exec(
                   select(Recipe)
                   .where(
                       col(Recipe.title).contains(q) |
                       col(Recipe.description).contains(q) |
                       col(Recipe.instructions).contains(q)
                   )
                   .limit(limit)
               ).all())

           return results
       ```

       - from eat_it.services.embedding import generate_embedding, serialize_f32
